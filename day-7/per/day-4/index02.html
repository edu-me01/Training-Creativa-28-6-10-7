3. Array Basics

Arrays store ordered collections of values.
// Create an array
const fruits = ['Apple', 'Banana', 'Mango'];
console.log(fruits[1]); // 'Banana'

// Add to an array
const tasks = [];
tasks.push('Study JavaScript');
console.log(tasks); // ['Study JavaScript']

4. Modifying Arrays

// Add and insert elements
const colors = ['red', 'blue'];
colors.push('green'); // ['red', 'blue', 'green']

// Insert at index 1
colors.splice(1, 0, 'yellow'); // ['red', 'yellow', 'blue', 'green']

5. Looping Through Arrays

// Using for...of
const tasks = ['Design', 'Code', 'Test'];
for (const task of tasks) {
  console.log(task);
}

// Using forEach
tasks.forEach((t, i) => console.log(`${i + 1}: ${t}`));

    7. localStorage Basics

    localStorage lets you save data in the browser.
    // Save and retrieve a value
localStorage.setItem('theme', 'dark');
console.log(localStorage.getItem('theme')); // 'dark'

// Save an object
const user = { name: 'Ali', level: 3 };
localStorage.setItem('user', JSON.stringify(user));

    8. localStorage with Arrays

    // Save and load an array
let tasks = JSON.parse(localStorage.getItem('tasks')) || [];
function saveTasks() {
  localStorage.setItem('tasks', JSON.stringify(tasks));
}
function addTask(text) {
  tasks.push({ text, done: false });
  saveTasks();
}

      10. Common Mistakes

      Always prevent default form submission and stringify objects for localStorage.
      // ❌ Will reload page if not prevented
form.addEventListener('submit', (e) => {
  e.preventDefault();
  // ...
});

// ❌ Forgetting to stringify
localStorage.setItem('data', { name: 'test' }); // Wrong
// ✅ Correct
localStorage.setItem('data', JSON.stringify({ name: 'test' }));

      11. Stats & Search

      // Get stats from tasks
function getStats(tasks) {
  return {
    total: tasks.length,
    done: tasks.filter(t => t.done).length,
    pending: tasks.filter(t => !t.done).length,
  };
}

// Search tasks by keyword
function searchTasks(keyword) {
  return tasks.filter(t => t.text.includes(keyword));
}
